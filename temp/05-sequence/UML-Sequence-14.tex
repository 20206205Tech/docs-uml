\subparagraph{Các thành phần tham gia}

\begin{table}[H]
\centering
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{|l|l|p{5cm}|}
\hline
\textbf{Thành phần} & \textbf{Phân loại} & \textbf{Vai trò trong hệ thống} \\ \hline
User & Actor & Người dùng thực hiện đặt lại mật khẩu mới. \\ \hline
Web Frontend & Component & Giao diện Next.js thu nhận mật khẩu mới và gửi yêu cầu đổi mật khẩu. \\ \hline
API Gateway (Kong) & Component & Định tuyến yêu cầu cập nhật thông tin người dùng. \\ \hline
Auth Service & Microservice & Dịch vụ Supabase Auth xác thực phiên và mã hóa lưu mật khẩu mới. \\ \hline
Database & Database & Lưu trữ băm mật khẩu mới của người dùng. \\ \hline
\end{tabular}
\caption{Các thành phần tham gia luồng Đặt lại / Đổi Mật khẩu}
\end{table}

\subparagraph{Mô tả chi tiết}
\begin{itemize}
\item Tại giao diện đổi mật khẩu, người dùng nhập mật khẩu mới.
\item Frontend gửi yêu cầu \texttt{PUT /auth/v1/user} kèm theo mật khẩu mới qua Gateway đến Auth Service (kèm Access Token tạm thời trong Header Authorization).
\item Auth Service xác minh token, băm mật khẩu mới và cập nhật trong Database, sau đó trả về bản ghi thành công để Frontend chuyển hướng người dùng vào trong hệ thống.
\end{itemize}
